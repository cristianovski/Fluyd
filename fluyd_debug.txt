        async function enviarSimulacao() {
            const tipo = document.getElementById('tipo_bem').value;
            const valor = document.getElementById('valor_credito').value;
            const btn = document.getElementById('btn-simular');

            if (!valor) { alert("Por favor, digite o valor que deseja conquistar."); return; }

            btn.innerText = "Conectando ao GEM..."; btn.style.opacity = "0.7"; btn.disabled = true;

            try {
                // SEU WEBHOOK AQUI
                await fetch('https://hook.us2.make.com/o93jlkarlvnk8kbw4emog88vutjqeh0v', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ tipo: tipo, valor: valor, data: new Date().toISOString() })
                });
                alert("✅ Simulação enviada! O GEM está processando.");
                btn.innerText = "Ver Parcelas";
            } catch (error) {
                console.error(error); alert("Erro de conexão. O GEM já foi notificado.");
            } finally {
                btn.style.opacity = "1"; btn.disabled = false;
            }
        }
